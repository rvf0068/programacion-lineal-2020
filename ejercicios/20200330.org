1. Resuelve el siguiente problema:
   \begin{equation*}
    \begin{aligned}
   \text{Maximizar} \quad & x_{1}+2x_{2}\\
   \text{sujeto a} \quad &
     \begin{aligned}
      x_{1}+3x_{2}+x_{3} &= 4\\
      -x_{1}+2x_{2}+x_{3} &= 2\\
       x_{1},x_{2},x_{3} &\geq 0
     \end{aligned}
   \end{aligned}
   \end{equation*}

2. Resuelve el siguiente problema:
   \begin{equation*}
    \begin{aligned}
   \text{Maximizar} \quad & x_{1}+2x_{2}\\
   \text{sujeto a} \quad &
     \begin{aligned}
      x_{1}+3x_{2}+x_{3} &= 4\\
      2x_{1}+2x_{2}+x_{3} &= 2\\
       x_{1},x_{2},x_{3} &\geq 0
     \end{aligned}
   \end{aligned}
   \end{equation*}

3. Resuelve el problema:
   \begin{equation*}
    \begin{aligned}
   \text{Maximizar} \quad & 3-\frac{1}{2}x_{1}+2x_{2}-\frac{3}{2}x_{4}\\
   \text{sujeto a} \quad &
     \begin{aligned}
      \frac{1}{2}x_{1}+x_{3}+\frac{1}{2}x_{4} &= 1\\
      -x_{1}+x_{2}-x_{4}+x_{5} &= 0\\
       x_{1}, x_{2},x_{3},x_{4},x_{5} & \geq 0
     \end{aligned}
   \end{aligned}
   \end{equation*}
